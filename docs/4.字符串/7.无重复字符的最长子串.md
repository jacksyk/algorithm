/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function (s) {
    let res = 0 // ç»“æœ
    let slow = 0

    for (let fast = 0; fast < s.length; fast++) {
        const cur = s[fast]
        const str = s.slice(slow, fast)
        const idx = str.indexOf(cur)
        if (idx !== -1) {
            slow += idx + 1
        }
        res = Math.max(res, fast - slow + 1)
    }
    return res
};
