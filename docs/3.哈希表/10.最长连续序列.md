var longestConsecutive = function (nums) {
    const set: Set<number> = new Set(nums)
    let res = 0
    for (let num of set.values()) {
        // psï¼šè¿™é‡Œè¿›è¡Œäº†å‰ªææ“ä½œ
        if (!set.has(num - 1)) {
            let count = 0
            while (set.has(num)) {
                count++
                num++
            }
            res = Math.max(res, count)
        }

    }
    return res
};
