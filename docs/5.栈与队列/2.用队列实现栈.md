class MyStack {
    queue: Array<number>
    constructor() {
        this.queue = []
    }

    push(x: number): void {
        this.queue.push(x)
    }

    pop(): number {
        return this.queue.pop() || -1
    }

    top(): number {
        return this.queue.concat([]).pop() || -1
    }

    empty(): boolean {
        return this.queue.length === 0
    }
}
