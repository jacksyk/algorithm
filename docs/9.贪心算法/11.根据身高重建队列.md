function reconstructQueue(people: number[][]): number[][] {
  people.sort((a, b) => {
    // èº«é«˜ç›¸ç­‰ï¼Œhå¤§çš„æ”¾åé¢
    if (a[0] === b[0]) {
      return a[1] - b[1];
    }
    return b[0] - a[0];
  });

  const res = people.concat();

  for (let i = 0; i < people.length; i++) {
    res.splice(i, 1);
    res.splice(people[i][1], 0, people[i]);
  }
  return res;
}
