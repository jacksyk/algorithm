function lemonadeChange(bills: number[]): boolean {
    const map = new Array(3).fill(0) // åˆ†åˆ«è¡¨ç¤º5å…ƒé’ç¥¨ï¼Œ10å…ƒé’ç¥¨ï¼Œ15å…ƒé’ç¥¨çš„å¼ æ•°
    for (let i = 0; i < bills.length; i++) {
        if (bills[i] === 20) {
            if (map[1] <= 0) {
                if (map[0] < 3) {
                    return false
                }
                map[0] -= 3
            } else {
                if (map[0] <= 0) {
                    return false
                }
                map[0]--
                map[1]--
                map[2]++
            }
        } else if (bills[i] === 10) {
            if (map[0] <= 0) {
                return false
            }
            map[1]++
            map[0]--
        } else {
            map[0]++
        }
    }
    return true
}
