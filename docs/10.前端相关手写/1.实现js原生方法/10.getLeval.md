Array.prototype.getLeval = function () {
    let depth = 1
    const dfs = (arr) => {
        for (let i = 0; i < arr.length; i++) {
            if (Array.isArray(arr[i])) {
                depth++
                dfs(arr[i])
            }
        }
    }
    dfs(this)   
    return depth
}
