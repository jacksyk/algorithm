Array.prototype.myFilter = function (callback) {
    return this.reduce((acc, cur, index) => {
        if (callback(cur, index)) {
            acc.push(cur)
        }
        return acc
    }, [])
}
let arr = [1, 2, 3, 4, 5]
arr = arr.myFilter((item) => {
    return item >= 3
})
