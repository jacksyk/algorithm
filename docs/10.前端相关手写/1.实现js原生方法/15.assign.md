Object.prototype._assign = function (target, ...source) {
    source.forEach((_obj) => {
        for (let key in _obj) {
            if (target.hasownProperty(key)) {
                target[key] = _obj[key]
            } else {
                target[key] = _obj[key]
            }
        }
    })
    return target
}
