function add(...arg1) {
    let totalSum = arg1.reduce((acc, cur) => acc + cur, 0)
    return function result(...arg2) {
        if (arg2.length === 0) {
            return totalSum
        } else {
            totalSum += arg2.reduce((acc, cur) => acc + cur, 0)
            return result
        }
    }
