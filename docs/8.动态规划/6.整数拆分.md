function integerBreak(n: number): number {
    const dp = new Array(n + 1).fill(0)
    dp[0] = 0
    dp[1] = 0
    dp[2] = 1
    for (let i = 3; i <= n; i++) {
        // æ‹†åˆ†æˆ0å…¶å®æ²¡æœ‰æ„ä¹‰ï¼Œæ‰€ä»¥è¿™é‡Œæ‹†åˆ†æˆ1å¼€å§‹
        for (let j = 1; j < i; j++) {
            dp[i] = Math.max(dp[i], j * (i - j), j * dp[i - j])
        }
    }
    return dp[n]
}
