    const initDp = () => {

        }
        for (let col = 0; col < n; col++) {
            dp[0][col] = 1
        }
    }
    dp[0][0] = 0
    initDp()

    for (let i = 1; i < m; i++) {

        }
    }

    return dp[m - 1][n - 1]
}
