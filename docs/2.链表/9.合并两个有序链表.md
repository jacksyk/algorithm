class ListNode {
  val: number;
  next: ListNode | null;
  constructor(val?: number, next?: null) {
    this.val = val === undefined ? 0 : val;
    this.next = next === undefined ? null : next;
  }

function mergeTwoLists(
  list1: ListNode | null,
  list2: ListNode | null

  let _vitrual = new ListNode(0); // è™šæ‹Ÿå¤´ç»“ç‚¹

    if (cur1.val < cur2.val) {
      _head.next = cur1;
      cur1 = cur1.next;
      _head = _head.next;
    } else {
      _head.next = cur2;
      cur2 = cur2.next;
      _head = _head.next;
    }

    _head.next = cur1;
    cur1 = cur1.next;
  }

    _head.next = cur2;
    cur2 = cur2.next;
  }
  return _vitrual.next;
}
