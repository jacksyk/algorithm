/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var subarraySum = function (nums, k) {
    let count = 0
    for (let i = 0; i < nums.length; i++) {
        let fast = i, sum = 0
        while (fast < nums.length) {
            sum += nums[fast]
            fast++
            if (sum === k) {
                count++
            }
        }
    }
    return count
};
